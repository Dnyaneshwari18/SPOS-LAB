package pagereplacement;

import java.util.*;

public class FIFOPageReplacement {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        // Ask user for number of frames
        System.out.print("Enter the number of frames: ");
        int capacity = sc.nextInt();
        
        // Ask user for the number of pages
        System.out.print("Enter the number of pages in reference string: ");
        int n = sc.nextInt();
        
        int[] pages = new int[n];
        System.out.println("Enter the page reference string (space-separated):");
        for (int i = 0; i < n; i++) {
            pages[i] = sc.nextInt();
        }
        
        Queue<Integer> frame = new LinkedList<>();
        Set<Integer> set = new HashSet<>();
        int faults = 0;
        
        System.out.println("FIFO Page Replacement Simulation");
        for (int page : pages) {
            if (!set.contains(page)) {
                if (frame.size() == capacity) {
                    int removed = frame.poll();
                    set.remove(removed);
                }
                frame.offer(page);
                set.add(page);
                faults++;
            }
            System.out.println("Frames after accessing " + page + ": " + frame);
        }
        System.out.println("Total Page Faults (FIFO): " + faults);
        sc.close();
    }
}
