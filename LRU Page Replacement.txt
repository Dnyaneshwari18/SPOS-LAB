package pagereplacement;

import java.util.*;

public class LRUPageReplacement {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        // Ask user for number of frames
        System.out.print("Enter the number of frames: ");
        int capacity = sc.nextInt();
        
        // Ask user for the number of pages
        System.out.print("Enter the number of pages in reference string: ");
        int n = sc.nextInt();
        
        int[] pages = new int[n];
        System.out.println("Enter the page reference string (space-separated):");
        for (int i = 0; i < n; i++) {
            pages[i] = sc.nextInt();
        }
        
        List<Integer> frame = new ArrayList<>();
        int faults = 0;
        
        System.out.println("LRU Page Replacement Simulation");
        for (int page : pages) {
            if (!frame.contains(page)) {
                if (frame.size() == capacity)
                    frame.remove(0);
                faults++;
            } else {
                frame.remove((Integer) page);
            }
            frame.add(page);
            System.out.println("Frames after accessing " + page + ": " + frame);
        }
        System.out.println("Total Page Faults (LRU): " + faults);
        sc.close();
    }
}
